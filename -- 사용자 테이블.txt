create DATABASE movie;
use movie;

-- 사용자 테이블
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(30) NOT NULL UNIQUE,      -- 아이디
    name VARCHAR(50) NOT NULL,                 -- 이름
    phone VARCHAR(13) NOT NULL UNIQUE,         -- 전화번호 (하이픈 포함)
    password VARCHAR(255) NOT NULL,            -- 비밀번호 (암호화된 형태)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 사용자 점수 필드 추가
ALTER TABLE users
ADD store_score INT DEFAULT 0,
ADD cinema_score INT DEFAULT 0;
ALTER TABLE users ADD COLUMN is_admin TINYINT(1) DEFAULT 0;

UPDATE users SET is_admin = 1 WHERE username = 'admin';


-- 공지사항 테이블
CREATE TABLE notices (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    author VARCHAR(100) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    views INT DEFAULT 0
);

-- DB 사용자 비밀번호 변경 (예시, 실제 비밀번호로 변경 필요)
ALTER USER 'db_user'@'localhost' IDENTIFIED WITH mysql_native_password BY '비밀번호';
FLUSH PRIVILEGES;

-- 상점 아이템 테이블
CREATE TABLE store_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price INT NOT NULL,
    components TEXT NOT NULL,
    purchase_limit INT NOT NULL,
    validity DATE NOT NULL,
    image_path VARCHAR(255)  -- 이미지 파일 경로 저장
);

-- 영화 테이블
CREATE TABLE movies (
    movie_id INT AUTO_INCREMENT PRIMARY KEY,       -- 영화 고유 ID
    title VARCHAR(255) NOT NULL,                   -- 제목
    description TEXT NOT NULL,                     -- 줄거리
    start_date DATE,                               -- 상영 시작일
    end_date DATE,                                 -- 상영 종료일
    director VARCHAR(255),                         -- 감독
    actors TEXT,                                   -- 출연진
    duration VARCHAR(50),                          -- 상영시간 (예: 120분)
    reviews TEXT,                                  -- 리뷰 또는 요약
    likes INT DEFAULT 0,                           -- 관심 수
    poster_path VARCHAR(255),                      -- 포스터 경로
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 영화 테이블에 명대사 컬럼 추가
ALTER TABLE movies ADD quote TEXT;

-- 영화관 테이블
CREATE TABLE cinemas (
    cinema_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,     -- 영화관 이름
    region VARCHAR(50) NOT NULL,    -- 지역 (예: 서울, 부산 등)
    address VARCHAR(255),
    contact VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 영화-영화관 연결 테이블
CREATE TABLE movie_cinemas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    movie_id INT NOT NULL,
    cinema_id INT NOT NULL,
    FOREIGN KEY (movie_id) REFERENCES movies(movie_id) ON DELETE CASCADE,
    FOREIGN KEY (cinema_id) REFERENCES cinemas(cinema_id) ON DELETE CASCADE
);

-- 영화 상영 일정 테이블
CREATE TABLE movie_showtimes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    movie_id INT NOT NULL,
    cinema_id INT NOT NULL,
    show_date DATE NOT NULL,       -- 날짜
    show_time TIME NOT NULL,       -- 상영 시간
    FOREIGN KEY (movie_id) REFERENCES movies(movie_id) ON DELETE CASCADE,
    FOREIGN KEY (cinema_id) REFERENCES cinemas(cinema_id) ON DELETE CASCADE
);

CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    movie_id INT NOT NULL,
    user_id INT NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (movie_id) REFERENCES movies(movie_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE movie_likes (
    like_id INT AUTO_INCREMENT PRIMARY KEY,           -- 고유 ID
    movie_id INT NOT NULL,                             -- 영화 ID
    user_id INT NOT NULL,                             -- 사용자 ID
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,   -- 좋아요 생성 시간
    FOREIGN KEY (movie_id) REFERENCES movies(movie_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    CONSTRAINT unique_movie_user UNIQUE (movie_id, user_id)  -- 동일한 영화에 대해 같은 사용자가 여러 번 좋아요를 누를 수 없게 제약 추가
);

CREATE TABLE reservations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    movie_id INT NOT NULL,
    cinema_id INT NOT NULL,
    show_date DATE NOT NULL,
    show_time TIME NOT NULL,
    seat_code VARCHAR(10) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (movie_id) REFERENCES movies(movie_id) ON DELETE CASCADE,
    FOREIGN KEY (cinema_id) REFERENCES cinemas(cinema_id) ON DELETE CASCADE
);

    name TEXT NOT NULL,
